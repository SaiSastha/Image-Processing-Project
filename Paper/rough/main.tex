\section{Our Model}

\begin{equation}
	u_{tt} + \gamma u_t = \operatorname{div}(c(|\nabla u|)\nabla u)
\end{equation}

The presence of $u_{tt}$ introduces wave-like (inertial) behaviour, damped by $\gamma u_t$.
This is a \textbf{hyperbolic} PDE (second-order in time).

\section*{Finite Difference Discretization}

Let $a = \Delta t$ (time step) and $h = \Delta x = \Delta y$ (spatial step).
Pixel at position $(i,j)$, time level $n$.

\subsection*{Time derivatives}

Central difference for $u_{tt}$:
\begin{equation}
	u_{tt}\big|_{i,j}^n = \frac{u_{i,j}^{n+1} - 2u_{i,j}^n + u_{i,j}^{n-1}}{a^2}
\end{equation}

Forward difference for $u_t$:
\begin{equation}
	\gamma u_t\big|_{i,j}^n = \gamma \cdot \frac{u_{i,j}^{n+1} - u_{i,j}^n}{a}
\end{equation}

\subsection*{Spatial diffusion term (RHS)}

Half-point diffusivity coefficients (arithmetic averaging):
\begin{equation}
	c_{i+\frac{1}{2},j} = \frac{c_{i+1,j}+c_{i,j}}{2}, \quad
	c_{i-\frac{1}{2},j} = \frac{c_{i-1,j}+c_{i,j}}{2}, \quad
	c_{i,j+\frac{1}{2}} = \frac{c_{i,j+1}+c_{i,j}}{2}, \quad
	c_{i,j-\frac{1}{2}} = \frac{c_{i,j}+c_{i,j-1}}{2}
\end{equation}

Discretized divergence:
\begin{equation}
	\operatorname{div}(c\nabla u)\big|_{i,j}^n =
	\frac{
		c_{i+\frac{1}{2},j}(u_{i+1,j}^n - u_{i,j}^n)
		- c_{i-\frac{1}{2},j}(u_{i,j}^n   - u_{i-1,j}^n)
	}{h^2}
	+
	\frac{
		c_{i,j+\frac{1}{2}}(u_{i,j+1}^n - u_{i,j}^n)
		- c_{i,j-\frac{1}{2}}(u_{i,j}^n   - u_{i,j-1}^n)
	}{h^2}
\end{equation}

\subsection*{Combined LHS and rearrangement}

Substituting into the PDE and multiplying through by $a^2$:
\begin{equation}
	u_{i,j}^{n+1}(1 + \gamma a) - (2 + \gamma a)\,u_{i,j}^n + u_{i,j}^{n-1}
	= a^2 \cdot \operatorname{div}(c\nabla u)\big|_{i,j}^n
\end{equation}

\textbf{Update formula} (solved for $u^{n+1}$):
\begin{equation}
	\boxed{
	u_{i,j}^{n+1} = \frac{a^2\cdot\operatorname{div}(c\nabla u)\big|_{i,j}^n
	+ (2 + \gamma a)\,u_{i,j}^n - u_{i,j}^{n-1}}{1 + \gamma a}
	}
\end{equation}

\nt{
	This is second-order in time: both $u^n$ \textbf{and} $u^{n-1}$ are needed at each stage.
	Initialization: $u^{-1} = u^0 = \tilde{u}$ (zero initial velocity assumption).
}

{\huge TNRD Discretization}

\section*{How TNRD is Discretized}

TNRD discretizes a gradient descent flow on a regularization energy. The continuous update is:
\begin{equation}
	u_t = -\nabla \mathcal{E}(u) = \sum_{k=1}^{K} f_k^T * \phi_k(f_k * u)
\end{equation}
An explicit Euler step with step size $\rho$ gives the $n$-th stage update:
\begin{equation}
	u^{(n+1)} = u^{(n)} + \rho^{(n)} \left[
	\sum_{k=1}^{K} \left(f_k^{(n)}\right)^T * \phi_k^{(n)}\!\left(f_k^{(n)} * u^{(n)}\right)
	- \lambda(u^{(n)} - \tilde{u})
	\right]
\end{equation}
where:
\begin{itemize}
	\item $f_k^{(n)}$ are stage-dependent learned convolutional filters,
	\item $\phi_k^{(n)}$ are stage-dependent learned influence functions (nonlinearities),
	\item $\lambda(u^{(n)} - \tilde{u})$ is the reaction/data fidelity term ($\tilde{u}$ = noisy input),
	\item $\rho^{(n)}$ is a learned step size.
\end{itemize}
The network is unrolled for $T$ stages (typically $T \leq 10$) and trained end-to-end or greedily.


\section*{Learnable Architecture from Our Model}

\subsection*{Discretization as a Learned Network}

From the explicit time discretization:
\begin{equation}
	u^{n+1} = \frac{1}{1 + \frac{\gamma\tau}{2}}\left[
	2u^n - \left(1 - \frac{\gamma\tau}{2}\right)u^{n-1}
	+ \tau^2\, \operatorname{div}(c(|\nabla u^n|)\nabla u^n)
	\right]
\end{equation}

We replace the diffusion operator $\operatorname{div}(c(|\nabla u|)\nabla u)$ with a learnable filter-bank approximation. Let $\{f_k\}_{k=1}^K$ be learned convolutional filters and $\phi_k$ fixed influence functions. Then the $n$-th stage update becomes:

\begin{equation}
	u^{(n+1)} = \frac{1}{1 + \frac{\gamma\tau}{2}}\left[
	2u^{(n)} - \left(1 - \frac{\gamma\tau}{2}\right)u^{(n-1)}
	+ \tau^2 \sum_{k=1}^{K} \left(f_k^{(n)}\right)^T * \phi_k\!\left(f_k^{(n)} * u^{(n)}\right)
	- \tau^2\lambda\left(u^{(n)} - \tilde{u}\right)
	\right]
\end{equation}

where:
\begin{itemize}
	\item $f_k^{(n)}$ are stage-dependent learned convolutional filters (as in TNRD),
	\item $\phi_k$ are \emph{fixed} influence functions (not learned, per our greedy training setup),
	\item $\gamma$ is fixed (not learned),
	\item $\lambda(u^{(n)} - \tilde{u})$ is the data fidelity term ($\tilde{u}$ = noisy input),
	\item $\tau$ is the time step (may be absorbed into learned parameters).
\end{itemize}

\nt{
	Each stage now requires both $u^{(n)}$ and $u^{(n-1)}$. At stage $n=0$, we initialize $u^{(-1)} = u^{(0)} = \tilde{u}$ (zero initial velocity).
}

\subsection*{Coefficient Comparison}
\begin{center}
	\begin{tabular}{lll}
		\hline
		\textbf{Term}              & \textbf{TNRD coefficient} & \textbf{Our Model coefficient}                \\
		\hline
		Current state $u^{(n)}$    & $1$                       & $\dfrac{2}{1 + \gamma\tau/2}$                 \\[6pt]
		Previous state $u^{(n-1)}$ & ---                       & $-\dfrac{1 - \gamma\tau/2}{1 + \gamma\tau/2}$ \\[6pt]
		Diffusion term             & $\rho^{(n)}$              & $\dfrac{\tau^2}{1 + \gamma\tau/2}$            \\[6pt]
		Data fidelity              & $-\rho^{(n)}\lambda$      & $-\dfrac{\tau^2\lambda}{1 + \gamma\tau/2}$    \\
		\hline
	\end{tabular}
\end{center}

The scalar $\alpha = \tfrac{1}{1+\gamma\tau/2}$ and $\beta = \tfrac{1-\gamma\tau/2}{1+\gamma\tau/2}$ can themselves be treated as learnable scalars (subject to $\alpha, \beta > 0$ and $\beta < 1$ for stability), while $\phi_k$ and $\gamma$ remain fixed.

\subsection*{Learned vs Fixed Parameters}

\begin{itemize}
	\item \textbf{Learned} (per stage $n$):
	      \begin{itemize}
		      \item $f_k^{(n)}$ — convolutional filters
		      \item $\alpha = \dfrac{1}{1+\gamma\tau/2}$, $\beta = \dfrac{1-\gamma\tau/2}{1+\gamma\tau/2}$ — momentum/damping scalars
		      \item $\lambda$ — data fidelity weight
	      \end{itemize}
	\item \textbf{Fixed}:
	      \begin{itemize}
		      \item $\phi_k$ — influence functions
		      \item $\gamma$ — damping coefficient
	      \end{itemize}
\end{itemize}

\nt{The filters $f_k^{(n)}$ carry the expressive power; $\alpha, \beta$ are just 2 scalars per stage and $\lambda$ a single scalar.}
